public class tamirinta2 {

    public static void main(String[] args) {

        //up ret
        for (int i = 1; i < 8; i++) {
            for (int j = i; j > 0; j--) {
                System.out.print(i);
            }
            System.out.println("");
        }
        //down ret
        for (int i = 6; i > 0; i--) {
            for (int j = i; j > 0; j--) {
                System.out.print(i);
            }
            System.out.println("");
        }

    }
}
